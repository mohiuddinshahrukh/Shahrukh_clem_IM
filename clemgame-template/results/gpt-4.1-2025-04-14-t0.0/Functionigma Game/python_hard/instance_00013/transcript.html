def edit_distance(a: str, b: str) -> int:
"""
Levenshtein edit distance (insert/delete/replace), O(len(a)*len(b)).
"""
n, m = len(a), len(b)
dp = list(range(m + 1))
for i in range(1, n + 1):
prev_diag = dp[0]
dp[0] = i
for j in range(1, m + 1):
temp = dp[j]
cost = 0 if a[i - 1] == b[j - 1] else 1
dp[j] = min(
dp[j] + 1, # delete
dp[j - 1] + 1, # insert
prev_diag + cost # replace
)

return dp[m]
